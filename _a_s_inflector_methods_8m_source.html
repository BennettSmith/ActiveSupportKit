00049 NSString *ASInflectorOrdinalize(NSNumber *number)
00050 {
00051     NSString *suffix;
00052     NSInteger absoluteIntegerValue = [number integerValue];
00053     // NSInteger is a long or an int depending on architecture. Avoid using abs,
00054     // labs or llabs from the standard C library to find out the absolute
00055     // integer value. Long integers are 64 bits wide in 64-bit software. Hence
00056     // the name and signature of the correct function changes according to
00057     // architecture. Let the compiler handle it. Just subtract from zero if
00058     // negative.
00059     if (absoluteIntegerValue < 0)
00060     {
00061         absoluteIntegerValue = -absoluteIntegerValue;
00062     }
00063     switch (absoluteIntegerValue % 100)
00064     {
00065         case 11:
00066         case 12:
00067         case 13:
00068             suffix = @"th";
00069             break;
00070         default:
00071         {
00072             switch (absoluteIntegerValue % 10)
00073             {
00074                 case 1:
00075                     suffix = @"st";
00076                     break;
00077                 case 2:
00078                     suffix = @"nd";
00079                     break;
00080                 case 3:
00081                     suffix = @"rd";
00082                     break;
00083                 default:
00084                     suffix = @"th";
00085             }
00086         }
00087     }
00088     return [[number description] stringByAppendingString:suffix];
00089 }

00091 NSString *ASInflectorApplyRulesAndReplacements(NSArray *rulesAndReplacements, NSString *word)
00092 {
00093     NSMutableString *result = [word mutableCopy];
00094     // Plurals is an array of arrays. The elements have two or three
00095     // sub-elements. If two elements, the first specifies a regular expression
00096     // and the second its replacement template. If three elements, the first
00097     // specifies a string target, the second its replacement and the third its
00098     // string compare options.
00099     //
00100     // Why is this necessary? Simply because Objective-C has no polymorphic
00101     // “gsub” method which accepts either a regular expression or string. Nor
00102     // can string-based search and replacements incorporate any flags;
00103     // comparison options amount to extra baggage. Hence the array of plurals
00104     // polymorphically carries different kinds of rules and their
00105     // replacements. The number of elements, the array count, determines
00106     // behaviour.
00107     for (NSArray *ruleAndReplacement in rulesAndReplacements)
00108     {
00109         if ([ruleAndReplacement count] == 2)
00110         {
00111             NSRegularExpression *re = [ruleAndReplacement objectAtIndex:0];
00112             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00113             if ([re replaceMatchesInString:result options:0 range:NSMakeRange(0, [result length]) withTemplate:replacement])
00114             {
00115                 break;
00116             }
00117         }
00118         else if ([ruleAndReplacement count] == 3)
00119         {
00120             NSString *target = [ruleAndReplacement objectAtIndex:0];
00121             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00122             NSUInteger options = [[ruleAndReplacement objectAtIndex:2] unsignedIntegerValue];
00123             if ([result replaceOccurrencesOfString:target withString:replacement options:options range:NSMakeRange(0, [result length])])
00124             {
00125                 break;
00126             }
00127         }
00128     }
00129     return [result copy];
00130 }
