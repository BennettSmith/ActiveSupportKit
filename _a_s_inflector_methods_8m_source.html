00028 NSString *ASInflectorCamelize(NSString *lowerCaseAndUnderscoredWord, BOOL firstLetterInUppercase)
00029 {
00030     NSString *camelizedString;
00031     if (firstLetterInUppercase)
00032     {
00033         camelizedString = [[NSRegularExpression regularExpressionWithPattern:@"/(.?)" options:0 error:NULL] replaceMatchesInString:lowerCaseAndUnderscoredWord replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
00034             return [@"::" stringByAppendingString:[[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"] uppercaseString]];
00035         }];
00036         //  (?:)    grouping without back-references
00037         //  ^|_     beginning of a line or string, or an underscore
00038         camelizedString = [[NSRegularExpression regularExpressionWithPattern:@"(?:^|_)(.)" options:0 error:NULL] replaceMatchesInString:camelizedString replacementStringForResult:^NSString *(NSTextCheckingResult *result, NSString *inString, NSInteger offset) {
00039             return [[[result regularExpression] replacementStringForResult:result inString:inString offset:offset template:@"$1"] uppercaseString];
00040         }];
00041     }
00042     else
00043     {
00044         camelizedString = ASInflectorCamelize(lowerCaseAndUnderscoredWord, YES);
00045         camelizedString = [[[lowerCaseAndUnderscoredWord substringWithRange:NSMakeRange(0, 1)] lowercaseString] stringByAppendingString:[camelizedString substringWithRange:NSMakeRange(1, [camelizedString length] - 1)]];
00046     }
00047     return camelizedString;

00069 NSString *ASInflectorOrdinalize(NSNumber *number)
00070 {
00071     NSString *suffix;
00072     NSInteger absoluteIntegerValue = [number integerValue];
00073     // NSInteger is a long or an int depending on architecture. Avoid using abs,
00074     // labs or llabs from the standard C library to find out the absolute
00075     // integer value. Long integers are 64 bits wide in 64-bit software. Hence
00076     // the name and signature of the correct function changes according to
00077     // architecture. Let the compiler handle it. Just subtract from zero if
00078     // negative.
00079     if (absoluteIntegerValue < 0)
00080     {
00081         absoluteIntegerValue = -absoluteIntegerValue;
00082     }
00083     switch (absoluteIntegerValue % 100)
00084     {
00085         case 11:
00086         case 12:
00087         case 13:
00088             suffix = @"th";
00089             break;
00090         default:
00091         {
00092             switch (absoluteIntegerValue % 10)
00093             {
00094                 case 1:
00095                     suffix = @"st";
00096                     break;
00097                 case 2:
00098                     suffix = @"nd";
00099                     break;
00100                 case 3:
00101                     suffix = @"rd";
00102                     break;
00103                 default:
00104                     suffix = @"th";
00105             }
00106         }
00107     }
00108     return [[number description] stringByAppendingString:suffix];
