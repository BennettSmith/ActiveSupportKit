00293 NSString *ASInflectorApplyRulesAndReplacements(NSArray *rulesAndReplacements, NSString *word)
00294 {
00295     NSMutableString *result = [word mutableCopy];
00296     // Plurals is an array of arrays. The elements have two or three
00297     // sub-elements. If two elements, the first specifies a regular expression
00298     // and the second its replacement template. If three elements, the first
00299     // specifies a string target, the second its replacement and the third its
00300     // string compare options.
00301     //
00302     // Why is this necessary? Simply because Objective-C has no polymorphic
00303     // “gsub” method which accepts either a regular expression or string. Nor
00304     // can string-based search and replacements incorporate any flags;
00305     // comparison options amount to extra baggage. Hence the array of plurals
00306     // polymorphically carries different kinds of rules and their
00307     // replacements. The number of elements, the array count, determines
00308     // behaviour.
00309     for (NSArray *ruleAndReplacement in rulesAndReplacements)
00310     {
00311         if ([ruleAndReplacement count] == 2)
00312         {
00313             NSRegularExpression *re = [ruleAndReplacement objectAtIndex:0];
00314             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00315             if ([re replaceMatchesInString:result options:0 range:NSMakeRange(0, [result length]) withTemplate:replacement])
00316             {
00317                 break;
00318             }
00319         }
00320         else if ([ruleAndReplacement count] == 3)
00321         {
00322             NSString *target = [ruleAndReplacement objectAtIndex:0];
00323             NSString *replacement = [ruleAndReplacement objectAtIndex:1];
00324             NSUInteger options = [[ruleAndReplacement objectAtIndex:2] unsignedIntegerValue];
00325             if ([result replaceOccurrencesOfString:target withString:replacement options:options range:NSMakeRange(0, [result length])])
00326             {
00327                 break;
00328             }
00329         }
00330     }
00331     return [result copy];
